# Personal Information Manager (Advanced Version)

def get_valid_input(prompt):
    user_input = input(prompt)
    while user_input.strip() == "":
        print("Input cannot be empty. Try again.")
        user_input = input(prompt)
    return user_input


def show_welcome():
    print("=" * 50)
    print("      PERSONAL INFORMATION MANAGER")
    print("=" * 50)
    print()


def show_menu():
    print("\nWhat would you like to do?")
    print("1. Enter your information")
    print("2. View your information")
    print("3. Exit")


def display_info(data):
    if not data:
        print("\nNo data available. Please enter your information first.")
        return

    print("\n" + "=" * 50)
    print("           YOUR INFORMATION")
    print("=" * 50)

    print(f"Name: {data['name']}")
    print(f"Age: {data['age']} years ({data['age'] * 12} months)")
    print(f"City: {data['city']}")
    print(f"Hobby: {data['hobby']}")
    print(f"Favorite Food: {data['food']}")
    print(f"Favorite Color: {data['color']}")

    print("=" * 50)


# Main program starts here
user_data = {}

show_welcome()

while True:
    show_menu()
    choice = input("Enter your choice (1-3): ")

    if choice == "1":
        print("\nEnter your details:")
        name = get_valid_input("Name: ")
        
        age_input = get_valid_input("Age: ")
        while not age_input.isdigit():
            print("Please enter a valid number.")
            age_input = get_valid_input("Age: ")
        age = int(age_input)

        city = get_valid_input("City: ")
        hobby = get_valid_input("Hobby: ")
        food = get_valid_input("Favorite Food: ")
        color = get_valid_input("Favorite Color: ")

        user_data = {
            "name": name,
            "age": age,
            "city": city,
            "hobby": hobby,
            "food": food,
            "color": color
        }

        print("\nInformation saved successfully!")

    elif choice == "2":
        display_info(user_data)

    elif choice == "3":
        print("\nGoodbye! Thanks for using the program.")
        break

    else:
        print("\nInvalid choice. Try again.")
